#!/bin/bash
# rbac-check.sh - Human-readable RBAC audit for OpenShift Virtualization groups

# ─── CONFIG ───────────────────────────────────────────────────────────────────
NAMESPACE="${NAMESPACE:-default}"
GREEN='\033[0;32m'
RED='\033[0;31m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
BOLD='\033[1m'
RESET='\033[0m'

# ─── GROUP → DISPLAY NAME MAPPING ─────────────────────────────────────────────
declare -A GROUP_LABELS=(
  ["support-engineering"]="Support Engineer"
  ["migration-operations"]="Migration Operations"
  ["platform-sre"]="Platform SRE"
  ["vm-read-operator"]="VM Read Operator"
)

# ─── PERMISSION MATRIX ────────────────────────────────────────────────────────
# Format: "display_message|verb|resource|subresource(optional)"
declare -A GROUP_CHECKS

GROUP_CHECKS["support-engineering"]="
view pods|get|pods|
read pod logs|get|pods|log
list deployments|list|deployments|
view services|get|services|
view configmaps|get|configmaps|
view virtual machines|get|virtualmachines|
read VM console logs|get|virtualmachineinstances|log
list namespaces|list|namespaces|
delete pods|delete|pods|
exec into pods|create|pods|exec
create virtual machines|create|virtualmachines|
delete virtual machines|delete|virtualmachines|
"

GROUP_CHECKS["migration-operations"]="
view pods|get|pods|
list virtual machines|list|virtualmachines|
view VM instances|get|virtualmachineinstances|
create VM migrations|create|virtualmachineinstancemigrations|
list VM migrations|list|virtualmachineinstancemigrations|
delete VM migrations|delete|virtualmachineinstancemigrations|
view datavolumes|get|datavolumes|
create datavolumes|create|datavolumes|
view persistent volume claims|get|persistentvolumeclaims|
delete virtual machines|delete|virtualmachines|
modify configmaps|update|configmaps|
view node resources|get|nodes|
"

GROUP_CHECKS["platform-sre"]="
view pods|get|pods|
delete pods|delete|pods|
exec into pods|create|pods|exec
view pod logs|get|pods|log
view configmaps|get|configmaps|
modify configmaps|update|configmaps|
view secrets|get|secrets|
list namespaces|list|namespaces|
view nodes|get|nodes|
cordon/uncordon nodes|patch|nodes|
list virtual machines|list|virtualmachines|
start/stop virtual machines|update|virtualmachines|
create VM migrations|create|virtualmachineinstancemigrations|
delete VM migrations|delete|virtualmachineinstancemigrations|
manage datavolumes|create|datavolumes|
view persistent volumes|get|persistentvolumes|
view cluster events|get|events|
manage rolebindings|create|rolebindings|
"

GROUP_CHECKS["vm-read-operator"]="
view virtual machines|get|virtualmachines|
list virtual machines|list|virtualmachines|
view VM instances|get|virtualmachineinstances|
list VM instances|list|virtualmachineinstances|
view VM migrations|get|virtualmachineinstancemigrations|
view datavolumes|get|datavolumes|
view persistent volume claims|get|persistentvolumeclaims|
view pods|get|pods|
read pod logs|get|pods|log
list services|list|services|
view configmaps|get|configmaps|
delete virtual machines|delete|virtualmachines|
create virtual machines|create|virtualmachines|
modify datavolumes|update|datavolumes|
exec into pods|create|pods|exec
delete pods|delete|pods|
"

# ─── FUNCTIONS ────────────────────────────────────────────────────────────────

check_permission() {
  local group="$1"
  local verb="$2"
  local resource="$3"
  local subresource="$4"

  if [[ -n "$subresource" ]]; then
    result=$(oc auth can-i "$verb" "$resource/$subresource" \
      --as-group="$group" --as="audit-user" -n "$NAMESPACE" 2>/dev/null)
  else
    result=$(oc auth can-i "$verb" "$resource" \
      --as-group="$group" --as="audit-user" -n "$NAMESPACE" 2>/dev/null)
  fi

  echo "$result"
}

print_group_header() {
  local label="$1"
  echo ""
  echo -e "${CYAN}${BOLD}╔══════════════════════════════════════════╗${RESET}"
  echo -e "${CYAN}${BOLD}   As a ${label}${RESET}"
  echo -e "${CYAN}${BOLD}╚══════════════════════════════════════════╝${RESET}"
  echo -e "  ${YELLOW}Namespace: ${NAMESPACE}${RESET}"
  echo ""
}

run_checks_for_group() {
  local group="$1"
  local label="${GROUP_LABELS[$group]}"
  local checks="${GROUP_CHECKS[$group]}"
  local pass=0
  local fail=0

  print_group_header "$label"

  while IFS= read -r line; do
    [[ -z "$line" ]] && continue

    IFS='|' read -r display_msg verb resource subresource <<< "$line"

    display_msg=$(echo "$display_msg" | xargs)
    verb=$(echo "$verb" | xargs)
    resource=$(echo "$resource" | xargs)
    subresource=$(echo "$subresource" | xargs)

    result=$(check_permission "$group" "$verb" "$resource" "$subresource")

    if [[ "$result" == "yes" ]]; then
      echo -e "  ${GREEN}✔${RESET}  I can ${display_msg}"
      ((pass++))
    else
      echo -e "  ${RED}✘${RESET}  As per role, I ${RED}cannot${RESET} ${display_msg}"
      ((fail++))
    fi

  done <<< "$checks"

  echo ""
  echo -e "  ${BOLD}Summary:${RESET} ${GREEN}${pass} allowed${RESET} | ${RED}${fail} denied${RESET}"
  echo ""
}

print_banner() {
  echo -e "${BOLD}"
  echo "  ╔═══════════════════════════════════════════════════╗"
  echo "  ║     OpenShift RBAC Human-Readable Audit Tool      ║"
  echo "  ╚═══════════════════════════════════════════════════╝"
  echo -e "${RESET}"
  echo -e "  ${YELLOW}Cluster:${RESET}    $(oc whoami --show-server 2>/dev/null)"
  echo -e "  ${YELLOW}Namespace:${RESET}  ${NAMESPACE}"
  echo -e "  ${YELLOW}Date:${RESET}       $(date '+%Y-%m-%d %H:%M:%S')"
}

# ─── OPTIONAL: OUTPUT TO FILE ─────────────────────────────────────────────────
save_report() {
  local outfile="rbac-audit-$(date +%Y%m%d-%H%M%S).txt"
  "$0" | sed 's/\x1b\[[0-9;]*m//g' > "$outfile"
  echo "Report saved to: $outfile"
}

# ─── USAGE ────────────────────────────────────────────────────────────────────
usage() {
  echo "Usage:"
  echo "  ./rbac-check.sh                          # Check all groups"
  echo "  ./rbac-check.sh --group platform-sre     # Check specific group"
  echo "  NAMESPACE=my-ns ./rbac-check.sh          # Check in specific namespace"
  echo "  ./rbac-check.sh --save                   # Save output to file"
}

# ─── MAIN ─────────────────────────────────────────────────────────────────────
main() {
  local target_group=""

  while [[ $# -gt 0 ]]; do
    case "$1" in
      --group) target_group="$2"; shift 2 ;;
      --save)  save_report; exit 0 ;;
      --help)  usage; exit 0 ;;
      *) echo "Unknown option: $1"; usage; exit 1 ;;
    esac
  done

  print_banner

  if [[ -n "$target_group" ]]; then
    if [[ -z "${GROUP_LABELS[$target_group]}" ]]; then
      echo "Unknown group: $target_group"
      echo "Available: ${!GROUP_LABELS[*]}"
      exit 1
    fi
    run_checks_for_group "$target_group"
  else
    for group in "${!GROUP_CHECKS[@]}"; do
      run_checks_for_group "$group"
    done
  fi
}

main "$@"
