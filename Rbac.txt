# ─── CORE CHECK FUNCTION ──────────────────────────────────────────────────────
check_permission() {
  local group="$1"
  local verb="$2"
  local resource="$3"
  local subresource="$4"

  if [[ -n "$subresource" ]]; then
    oc auth can-i "$verb" "$resource/$subresource" \
      --as-group="$group" --as="audit-user" -n "$NAMESPACE" 2>/dev/null
  else
    oc auth can-i "$verb" "$resource" \
      --as-group="$group" --as="audit-user" -n "$NAMESPACE" 2>/dev/null
  fi
}

# ─── SEVERITY BADGE ──────────────────────────────────────────────────────────
severity_badge() {
  case "$1" in
    HIGH)   echo -e "${RED}[HIGH]${RESET}" ;;
    MEDIUM) echo -e "${YELLOW}[MEDIUM]${RESET}" ;;
    LOW)    echo -e "${CYAN}[LOW]${RESET}" ;;
    *)      echo -e "${YELLOW}[UNKNOWN]${RESET}" ;;
  esac
}

# ─── PRINT SECTION HEADER ─────────────────────────────────────────────────────
print_group_header() {
  local label="$1"
  local group="$2"
  echo ""
  echo -e "${CYAN}${BOLD}╔══════════════════════════════════════════════════════════╗${RESET}"
  printf "${CYAN}${BOLD}  %-56s  ${RESET}\n" "  👤 As a ${label} (${group})"
  echo -e "${CYAN}${BOLD}╚══════════════════════════════════════════════════════════╝${RESET}"
  echo -e "  ${YELLOW}Namespace: ${NAMESPACE}${RESET}"
}

# ─── EXPECTED PERMISSIONS SECTION ────────────────────────────────────────────
run_expected_checks() {
  local group="$1"
  local checks="${GROUP_CHECKS[$group]}"
  local pass=0 fail=0

  echo ""
  echo -e "  ${BOLD}── Expected Capabilities ──────────────────────────────────${RESET}"
  echo ""

  while IFS= read -r line; do
    [[ -z "$line" ]] && continue
    IFS='|' read -r display_msg verb resource subresource <<< "$line"
    display_msg=$(echo "$display_msg" | xargs)
    verb=$(echo "$verb" | xargs)
    resource=$(echo "$resource" | xargs)
    subresource=$(echo "$subresource" | xargs)

    result=$(check_permission "$group" "$verb" "$resource" "$subresource")

    if [[ "$result" == "yes" ]]; then
      echo -e "  ${GREEN}✔${RESET}  I can ${display_msg}"
      ((pass++))
    else
      echo -e "  ${RED}✘${RESET}  As per role, I ${RED}cannot${RESET} ${display_msg}"
      ((fail++))
    fi
  done <<< "$checks"

  echo ""
  echo -e "  ${BOLD}Result:${RESET} ${GREEN}${pass} granted${RESET} | ${RED}${fail} missing${RESET}"

  # Return counts via globals for summary
  EXPECTED_PASS=$pass
  EXPECTED_FAIL=$fail
}

# ─── ANOMALY DETECTION SECTION ───────────────────────────────────────────────
run_anomaly_checks() {
  local group="$1"
  local checks="${ANOMALY_CHECKS[$group]}"
  local anomaly_count=0
  local clean_count=0
  local anomaly_lines=()

  while IFS= read -r line; do
    [[ -z "$line" ]] && continue
    IFS='|' read -r display_msg verb resource subresource severity <<< "$line"
    display_msg=$(echo "$display_msg" | xargs)
    verb=$(echo "$verb" | xargs)
    resource=$(echo "$resource" | xargs)
    subresource=$(echo "$subresource" | xargs)
    severity=$(echo "$severity" | xargs)

    result=$(check_permission "$group" "$verb" "$resource" "$subresource")

    if [[ "$result" == "yes" ]]; then
      badge=$(severity_badge "$severity")
      anomaly_lines+=("  ${MAGENTA}⚠${RESET}  ${badge} I can ${BOLD}${display_msg}${RESET} — ${YELLOW}this is outside expected role scope${RESET}")
      ((anomaly_count++))
    else
      ((clean_count++))
    fi
  done <<< "$checks"

  echo ""
  echo -e "  ${BOLD}── Privilege Anomaly Report ────────────────────────────────${RESET}"
  echo ""

  if [[ ${#anomaly_lines[@]} -eq 0 ]]; then
    echo -e "  ${GREEN}✔  No unexpected privileges detected. Role scope looks clean.${RESET}"
  else
    for entry in "${anomaly_lines[@]}"; do
      echo -e "$entry"
    done
  fi

  echo ""
  echo -e "  ${BOLD}Result:${RESET} ${MAGENTA}${anomaly_count} anomalies found${RESET} | ${GREEN}${clean_count} checks clean${RESET}"

  ANOMALY_COUNT=$anomaly_count
}

# ─── GROUP SUMMARY ────────────────────────────────────────────────────────────
print_group_footer() {
  local label="$1"
  local exp_pass=$2
  local exp_fail=$3
  local anomalies=$4

  echo ""
  echo -e "  ${BOLD}═══ ${label} Overall Summary ═══${RESET}"
  echo -e "  ${GREEN}✔  Expected granted  : ${exp_pass}${RESET}"
  echo -e "  ${RED}✘  Expected missing  : ${exp_fail}${RESET}"
  echo -e "  ${MAGENTA}⚠  Anomalies found   : ${anomalies}${RESET}"

  if [[ $anomalies -gt 0 ]]; then
    echo -e "  ${YELLOW}→  Review ArgoCD ClusterRole/RoleBinding for this group!${RESET}"
  fi
  echo ""
}

# ─── FINAL AUDIT SUMMARY ──────────────────────────────────────────────────────
declare -A SUMMARY_PASS
declare -A SUMMARY_FAIL
declare -A SUMMARY_ANOMALY

print_final_summary() {
  echo ""
  echo -e "${BOLD}${CYAN}╔══════════════════════════════════════════════════════════╗${RESET}"
  echo -e "${BOLD}${CYAN}              CLUSTER-WIDE RBAC AUDIT SUMMARY              ${RESET}"
  echo -e "${BOLD}${CYAN}╚══════════════════════════════════════════════════════════╝${RESET}"
  echo ""
  printf "  ${BOLD}%-30s %10s %10s %12s${RESET}\n" "Group" "Granted" "Missing" "Anomalies"
  echo "  ──────────────────────────────────────────────────────────"

  local total_anomalies=0
  for group in "${!GROUP_LABELS[@]}"; do
    local label="${GROUP_LABELS[$group]}"
    local p="${SUMMARY_PASS[$group]:-0}"
    local f="${SUMMARY_FAIL[$group]:-0}"
    local a="${SUMMARY_ANOMALY[$group]:-0}"
    ((total_anomalies += a))

    local anomaly_display="${a}"
    if [[ $a -gt 0 ]]; then
      anomaly_display="${MAGENTA}${a} ⚠${RESET}"
    else
      anomaly_display="${GREEN}${a} ✔${RESET}"
    fi

    printf "  %-30s ${GREEN}%10s${RESET} ${RED}%10s${RESET} " "$label" "$p" "$f"
    echo -e "$anomaly_display"
  done

  echo ""
  if [[ $total_anomalies -gt 0 ]]; then
    echo -e "  ${RED}${BOLD}⚠  ACTION REQUIRED: ${total_anomalies} anomalous privilege(s) detected across groups.${RESET}"
    echo -e "  ${YELLOW}   Review your ArgoCD RBAC manifests and check ClusterRole aggregation rules.${RESET}"
  else
    echo -e "  ${GREEN}${BOLD}✔  All groups are within expected privilege boundaries.${RESET}"
  fi
  echo ""
}

# ─── BANNER ───────────────────────────────────────────────────────────────────
print_banner() {
  echo -e "${BOLD}"
  echo "  ╔══════════════════════════════════════════════════════════╗"
  echo "  ║    OpenShift RBAC Audit + Privilege Anomaly Detector     ║"
  echo "  ╚══════════════════════════════════════════════════════════╝"
  echo -e "${RESET}"
  echo -e "  ${YELLOW}Cluster  :${RESET}  $(oc whoami --show-server 2>/dev/null)"
  echo -e "  ${YELLOW}Namespace:${RESET}  ${NAMESPACE}"
  echo -e "  ${YELLOW}Date     :${RESET}  $(date '+%Y-%m-%d %H:%M:%S')"
  echo ""
  echo -e "  ${CYAN}Sections per group:${RESET}"
  echo -e "  ${GREEN}✔${RESET} Expected Capabilities  — what this role should be able to do"
  echo -e "  ${MAGENTA}⚠${RESET} Privilege Anomalies    — what this role should NEVER do"
  echo ""
}

# ─── SAVE REPORT ──────────────────────────────────────────────────────────────
save_report() {
  local outfile="rbac-audit-$(date +%Y%m%d-%H%M%S).txt"
  # Re-run self, strip ANSI codes, save
  NAMESPACE="$NAMESPACE" bash "$0" --all 2>/dev/null \
    | sed 's/\x1b\[[0-9;]*m//g' > "$outfile"
  echo -e "  ${GREEN}Report saved to: ${outfile}${RESET}"
}

# ─── RUN FOR ONE GROUP ────────────────────────────────────────────────────────
run_group() {
  local group="$1"
  local label="${GROUP_LABELS[$group]}"

  print_group_header "$label" "$group"
  run_expected_checks "$group"
  run_anomaly_checks "$group"
  print_group_footer "$label" "$EXPECTED_PASS" "$EXPECTED_FAIL" "$ANOMALY_COUNT"

  SUMMARY_PASS[$group]=$EXPECTED_PASS
  SUMMARY_FAIL[$group]=$EXPECTED_FAIL
  SUMMARY_ANOMALY[$group]=$ANOMALY_COUNT
}

# ─── USAGE ────────────────────────────────────────────────────────────────────
usage() {
  echo ""
  echo "Usage:"
  echo "  ./rbac-audit.sh                           # Audit all groups"
  echo "  ./rbac-audit.sh --group platform-sre      # Audit one group"
  echo "  NAMESPACE=my-ns ./rbac-audit.sh           # Specify namespace"
  echo "  ./rbac-audit.sh --save                    # Save output to file"
  echo ""
  echo "Available groups:"
  for g in "${!GROUP_LABELS[@]}"; do
    echo "  $g  →  ${GROUP_LABELS[$g]}"
  done
  echo ""
}

# ─── MAIN ─────────────────────────────────────────────────────────────────────
main() {
  local target_group=""
  local do_save=false

  while [[ $# -gt 0 ]]; do
    case "$1" in
      --group) target_group="$2"; shift 2 ;;
      --save)  do_save=true; shift ;;
      --all)   shift ;;  # internal flag used by --save
      --help)  usage; exit 0 ;;
      *) echo "Unknown option: $1"; usage; exit 1 ;;
    esac
  done

  print_banner

  if [[ "$do_save" == true ]]; then
    save_report
    exit 0
  fi

  if [[ -n "$target_group" ]]; then
    if [[ -z "${GROUP_LABELS[$target_group]}" ]]; then
      echo "Unknown group: $target_group"
      echo "Available: ${!GROUP_LABELS[*]}"
      exit 1
    fi
    run_group "$target_group"
  else
    for group in "support-engineering" "migration-operations" "platform-sre" "vm-read-operator"; do
      run_group "$group"
    done
    print_final_summary
  fi
}

main "$@"
